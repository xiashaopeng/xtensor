170 template <class E1, class E2>
171 inline auto searchsorted(E1&& a, E2&& v, bool right = true)
172 {
173 XTENSOR_ASSERT(std::is_sorted(a.cbegin(), a.cend()));
174
175 auto out = xt::empty<size_t>(v.shape());
176
177 if (right)
178 {
179 for (size_t i = 0; i < v.size(); ++i)
180 {
181 out(i) = static_cast<std::size_t>(std::lower_bound(a.cbegin(), a.cend(), v(i)) - a.cbegin());
182 }
183 }
184 else
185 {
186 for (size_t i = 0; i < v.size(); ++i)
187 {
188 out(i) = static_cast<std::size_t>(std::upper_bound(a.cbegin(), a.cend(), v(i)) - a.cbegin());
189 }
190 }
191
192
193 return out;
194 }
