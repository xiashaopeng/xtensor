199 template <class T, class E, class D>
200 struct random_impl
201 {
202 using value_type = T;
203
204 random_impl(E& engine, D&& dist)
205 : m_engine(engine), m_dist(std::move(dist))
206 {
207 }
208
209 template <class... Args>
210 inline value_type operator()(Args...) const
211 {
212 return m_dist(m_engine);
213 }
214
215 template <class It>
216 inline value_type element(It, It) const
217 {
218 return m_dist(m_engine);
219 }
220
221 template <class EX>
222 inline void assign_to(xexpression<EX>& e) const noexcept
223 {
224 // Note: we're not going row/col major here
225 auto& ed = e.derived_cast();
226 for (auto&& el : ed.storage())
227 {
228 el = m_dist(m_engine);
229 }
230 }
231
232 private:
233
234 E& m_engine;
235 mutable D m_dist;
236 };

685 template <class T, class E>
686 void shuffle(xexpression<T>& e, E& engine)
687 {
688 T& de = e.derived_cast();
689
690 if (de.dimension() == 1)
691 {
692 using size_type = typename T::size_type;
693 auto first = de.begin();
694 auto last = de.end();
695
696 for (size_type i = std::size_t((last - first) - 1); i > 0; --i)
697 {
698 std::uniform_int_distribution<size_type> dist(0, i);
699 auto j = dist(engine);
700 using std::swap;
701 swap(first[i], first[j]);
702 }
703 }
704 else
705 {
706 using size_type = typename T::size_type;
707 decltype(auto) buf = empty_like(view(de, 0));
708
709 for (size_type i = de.shape()[0] - 1; i > 0; --i)
710 {
711 std::uniform_int_distribution<size_type> dist(0, i);
712 size_type j = dist(engine);
713
714 buf = view(de, j);
715 view(de, j) = view(de, i);
716 view(de, i) = buf;
717 }
718 }
719 }

766 template <class T, class E>
767 xtensor<typename T::value_type, 1> choice(const xexpression<T>& e, std::size_t n, bool replace, E& engine)
768 {
769 const auto& de = e.derived_cast();
770 XTENSOR_ASSERT((de.dimension() == 1));
771 XTENSOR_ASSERT((replace || n <= de.size()));
772 using result_type = xtensor<typename T::value_type, 1>;
773 using size_type = typename result_type::size_type;
774 result_type result;
775 result.resize({n});
776
777 if (replace)
778 {
779 auto dist = std::uniform_int_distribution<size_type>(0, de.size() - 1);
780 for (size_type i = 0; i < n; ++i)
781 {
782 result[i] = de.storage()[dist(engine)];
783 }
784 }
785 else
786 {
787 // Naive resevoir sampling without weighting:
788 std::copy(de.storage().begin(), de.storage().begin() + n, result.begin());
789 size_type i = n;
790 for(auto it = de.storage().begin() + n; it != de.storage().end(); ++it, ++i)
791 {
792 auto idx = std::uniform_int_distribution<size_type>(0, i)(engine);
793 if (idx < n)
794 {
795 result.storage()[idx] = *it;
796 }
797 }
798 }
799 return result;
800 }

828 template <class T, class W, class E>
829 xtensor<typename T::value_type, 1>
830 choice(const xexpression<T>& e, std::size_t n, const xexpression<W>& weights, bool replace, E& engine)
831 {
832 const auto& de = e.derived_cast();
833 const auto& dweights = weights.derived_cast();
834 XTENSOR_ASSERT((de.dimension() == 1));
835 XTENSOR_ASSERT((replace || n <= de.size()));
836 XTENSOR_ASSERT((de.size() == dweights.size()));
