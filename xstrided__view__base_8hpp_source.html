819 template <class adj_strides_policy>
820 struct strided_view_args : adj_strides_policy
821 {
822 using base_type = adj_strides_policy;
823
824 template <class S, class ST, class V>
825 void fill_args(const S& shape, ST&& old_strides, std::size_t base_offset, layout_type layout, const V& slices)
826 {
827 // Compute dimension
828 std::size_t dimension = shape.size(), n_newaxis = 0, n_add_all = 0;
829 std::ptrdiff_t dimension_check = static_cast<std::ptrdiff_t>(shape.size());
830
831 bool has_ellipsis = false;
832 for (const auto& el : slices)
833 {
834 if (xtl::get_if<xt::xnewaxis_tag>(&el) != nullptr)
835 {
836 ++dimension;
837 ++n_newaxis;
838 }
839 else if (xtl::get_if<std::ptrdiff_t>(&el) != nullptr)
840 {
841 --dimension;
842 --dimension_check;
843 }
844 else if (xtl::get_if<xt::xellipsis_tag>(&el) != nullptr)
845 {
846 if (has_ellipsis == true)
847 {
848 XTENSOR_THROW(std::runtime_error, "Ellipsis can only appear once.");
849 }
850 has_ellipsis = true;
851 }
852 else
853 {
854 --dimension_check;
855 }
856 }
857
858 if (dimension_check < 0)
859 {
860 XTENSOR_THROW(std::runtime_error, "Too many slices for view.");
861 }
862
863 if (has_ellipsis)
864 {
865 // replace ellipsis with N * xt::all
866 // remove -1 because of the ellipsis slize itself
867 n_add_all = shape.size() - (slices.size() - 1 - n_newaxis);
868 }
869
870 // Compute strided view
871 new_offset = base_offset;
872 new_shape.resize(dimension);
873 new_strides.resize(dimension);
874 base_type::resize(dimension);
875
876 auto old_shape = shape;
877 using old_strides_value_type = typename std::decay_t<ST>::value_type;
878
879 std::ptrdiff_t axis_skip = 0;
880 std::size_t idx = 0, i = 0, i_ax = 0;
881
882 auto slice_getter = detail::slice_getter_impl<S>(shape);
883
884 for (; i < slices.size(); ++i)
885 {
886 i_ax = static_cast<std::size_t>(static_cast<std::ptrdiff_t>(i) - axis_skip);
887 auto ptr = xtl::get_if<std::ptrdiff_t>(&slices[i]);
888 if (ptr != nullptr)
889 {
890 auto slice0 = static_cast<old_strides_value_type>(*ptr);
891 new_offset += static_cast<std::size_t>(slice0 * old_strides[i_ax]);
892 }
893 else if (xtl::get_if<xt::xnewaxis_tag>(&slices[i]) != nullptr)
894 {
895 new_shape[idx] = 1;
896 base_type::set_fake_slice(idx);
897 ++axis_skip, ++idx;
898 }
899 else if (xtl::get_if<xt::xellipsis_tag>(&slices[i]) != nullptr)
900 {
901 for (std::size_t j = 0; j < n_add_all; ++j)
902 {
903 new_shape[idx] = old_shape[i_ax];
904 new_strides[idx] = old_strides[i_ax];
905 base_type::set_fake_slice(idx);
906 ++idx, ++i_ax;
907 }
908 axis_skip = axis_skip - static_cast<std::ptrdiff_t>(n_add_all) + 1;
909 }
910 else if (xtl::get_if<xt::xall_tag>(&slices[i]) != nullptr)
911 {
912 new_shape[idx] = old_shape[i_ax];
913 new_strides[idx] = old_strides[i_ax];
914 base_type::set_fake_slice(idx);
915 ++idx;
916 }
917 else if (base_type::fill_args(slices, i, idx,
918 old_shape[i_ax],
919 old_strides[i_ax],
920 new_shape, new_strides))
921 {
922 ++idx;
923 }
924 else
925 {
926 slice_getter.idx = i_ax;
927 auto info = xtl::visit(slice_getter, slices[i]);
928 new_offset += static_cast<std::size_t>(info[0] * old_strides[i_ax]);
929 new_shape[idx] = static_cast<std::size_t>(info[1]);
930 new_strides[idx] = info[2] * old_strides[i_ax];
931 base_type::set_fake_slice(idx);
932 ++idx;
933 }
934 }
935
936 i_ax = static_cast<std::size_t>(static_cast<std::ptrdiff_t>(i) - axis_skip);
937 for (; i_ax < old_shape.size(); ++i_ax, ++idx)
938 {
939 new_shape[idx] = old_shape[i_ax];
940 new_strides[idx] = old_strides[i_ax];
941 base_type::set_fake_slice(idx);
942 }
943
944 new_layout = do_strides_match(new_shape, new_strides, layout, true) ? layout : layout_type::dynamic;
945 }
946
947 using shape_type = dynamic_shape<std::size_t>;
948 shape_type new_shape;
949 using strides_type = dynamic_shape<std::ptrdiff_t>;
950 strides_type new_strides;
951 std::size_t new_offset;
952 layout_type new_layout;
953 };
