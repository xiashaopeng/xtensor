246 // Overload isinf, isnan and isfinite for complex datatypes,
247 // following the Python specification:
248 template <class T>
249 inline bool isinf(const std::complex<T>& c)
250 {
251 return std::isinf(std::real(c)) || std::isinf(std::imag(c));
252 }
253
254 template <class T>
255 inline bool isnan(const std::complex<T>& c)
256 {
257 return std::isnan(std::real(c)) || std::isnan(std::imag(c));
258 }
259
260 template <class T>
261 inline bool isfinite(const std::complex<T>& c)
262 {
263 return !isinf(c) && !isnan(c);

368#define XTENSOR_REDUCER_FUNCTION(NAME, FUNCTOR, INIT_VALUE_TYPE, INIT) \
369 template <class T = void, class E, class X, class EVS = DEFAULT_STRATEGY_REDUCERS, \
370 XTL_REQUIRES(xtl::negation<is_reducer_options<X>>, xtl::negation<xtl::is_integral<std::decay_t<X> >> )> \
371 inline auto NAME(E&& e, X&& axes, EVS es = EVS()) \
372 { \
373 using init_value_type = std::conditional_t<std::is_same<T, void>::value, INIT_VALUE_TYPE, T>; \
374 using functor_type = FUNCTOR; \
375 using init_value_fct = xt::const_value<init_value_type>; \
376 return xt::reduce(make_xreducer_functor(functor_type(), \
377 init_value_fct(detail::fill_init<init_value_type>(INIT))), \
378 std::forward<E>(e), \
379 std::forward<X>(axes), es); \
380 } \
381 \
382 template <class T = void, class E, class X, class EVS = DEFAULT_STRATEGY_REDUCERS, \
383 XTL_REQUIRES(xtl::negation<is_reducer_options<X>>, xtl::is_integral<std::decay_t<X>>)> \
384 inline auto NAME(E&& e, X axis, EVS es = EVS()) \
385 { \
386 return NAME(std::forward<E>(e), {axis}, es); \
387 } \
388 \
389 template <class T = void, class E, class EVS = DEFAULT_STRATEGY_REDUCERS, \
390 XTL_REQUIRES(is_reducer_options<EVS>)> \
391 inline auto NAME(E&& e, EVS es = EVS()) \
392 { \
393 using init_value_type = std::conditional_t<std::is_same<T, void>::value, INIT_VALUE_TYPE, T>; \
394 using functor_type = FUNCTOR; \
395 using init_value_fct = xt::const_value<init_value_type >; \
396 return xt::reduce(make_xreducer_functor(functor_type(), \
397 init_value_fct(detail::fill_init<init_value_type >(INIT))), std::forward<E>(e), es); \
398 } \

403 using init_value_type = std::conditional_t<std::is_same<T, void>::value, INIT_VALUE_TYPE, T>; \

1761 {
1762 using internal_type = xtl::promote_type_t<A1, A2, double>;
1763 if (math::isnan(a) && math::isnan(b))
1764 {
1765 return m_equal_nan;
1766 }
1767 if (math::isinf(a) && math::isinf(b))
1768 {

1770 return a == b;
1771 }

1773 return d <= m_atol || d <= m_rtol * double((std::max)(math::abs(internal_type(a)), math::abs(internal_type(b))));

1969 template <class T = void, class E, class W, class X, class EVS = DEFAULT_STRATEGY_REDUCERS,
1970 XTL_REQUIRES(is_reducer_options<EVS>, xtl::negation<xtl::is_integral<X>>)>
1971 inline auto average(E&& e, W&& weights, X&& axes, EVS ev = EVS())
1972 {
1973 xindex_type_t<typename std::decay_t<E>::shape_type> broadcast_shape;
1974 xt::resize_container(broadcast_shape, e.dimension());
1975 auto ax = normalize_axis(e, axes);
1976 if (weights.dimension() == 1)
1977 {
1978 if (weights.size() != e.shape()[ax[0]])
1979 {
1980 XTENSOR_THROW(std::runtime_error, "Weights need to have the same shape as expression at axes.");
1981 }
1982
1983 std::fill(broadcast_shape.begin(), broadcast_shape.end(), std::size_t(1));
1984 broadcast_shape[ax[0]] = weights.size();
1985 }
1986 else
1987 {
