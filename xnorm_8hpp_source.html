204#define XTENSOR_DEFINE_SIGNED_NORMS(T) \
205 inline auto \
206 norm_lp(T t, double p) noexcept \
207 { \
208 using rt = decltype(std::abs(t)); \
209 return p == 0.0 \
210 ? static_cast<rt>(t != 0) \
211 : std::abs(t); \
212 } \
213 inline auto \
214 norm_lp_to_p(T t, double p) noexcept \
215 { \
216 using rt = xtl::real_promote_type_t<T>; \
217 return p == 0.0 \
218 ? static_cast<rt>(t != 0) \
219 : std::pow(static_cast<rt>(std::abs(t)), \
220 static_cast<rt>(p)); \
221 } \
222 inline std::size_t norm_l0(T t) noexcept { return (t != 0); } \
223 inline auto norm_l1(T t) noexcept { return std::abs(t); } \
224 inline auto norm_l2(T t) noexcept { return std::abs(t); } \
225 inline auto norm_linf(T t) noexcept { return std::abs(t); } \
226 inline auto norm_sq(T t) noexcept { return t * t; }

277 template <class T>
278 inline uint64_t norm_l0(const std::complex<T>& t) noexcept
279 {
280 return t.real() != 0 || t.imag() != 0;
281 }
282
286 template <class T>
287 inline auto norm_l1(const std::complex<T>& t) noexcept
288 {
289 return std::abs(t.real()) + std::abs(t.imag());
290 }
291
296 template <class T>
297 inline auto norm_l2(const std::complex<T>& t) noexcept
298 {
299 return std::abs(t);
300 }
301
307 template <class T>
308 inline auto norm_sq(const std::complex<T>& t) noexcept
309 {
310 // Does not use std::norm since it returns a std::complex on OSX
311 return t.real() * t.real() + t.imag() * t.imag();
312 }
313
317 template <class T>
318 inline auto norm_linf(const std::complex<T>& t) noexcept
319 {
320 return (std::max)(std::abs(t.real()), std::abs(t.imag()));
321 }
322
326 template <class T>
327 inline auto norm_lp_to_p(const std::complex<T>& t, double p) noexcept
328 {
329 using rt = decltype(std::pow(std::abs(t.real()), static_cast<T>(p)));
330 return p == 0
331 ? static_cast<rt>(t.real() != 0 || t.imag() != 0)
332 : std::pow(std::abs(t.real()), static_cast<T>(p)) +
333 std::pow(std::abs(t.imag()), static_cast<T>(p));
334 }
335
339 template <class T>
340 inline auto norm_lp(const std::complex<T>& t, double p) noexcept
341 {
342 return p == 0
343 ? norm_lp_to_p(t, p)
344 : std::pow(norm_lp_to_p(t, p), 1.0 / p);
345 }
