247 template <class T, class S>
248 inline void nested_optional_copy(T&& iter, const S& s)
249 {
250 iter->value() = s.value();
251 iter->has_value() = s.has_value();
252 ++iter;
253 }
254
255 template <class T, class S>
256 inline void nested_optional_copy(T&& iter, std::initializer_list<S> s)
257 {
258 for (auto it = s.begin(); it != s.end(); ++it)
259 {
260 nested_optional_copy(std::forward<T>(iter), *it);
261 }
262 }
