548 template <class S1, class S2>
549 inline void xchunked_array<CS>::resize(S1&& shape, S2&& chunk_shape, layout_type chunk_memory_layout)
550 {
551 // compute chunk number in each dimension (shape_of_chunks)
552 std::vector<std::size_t> shape_of_chunks(shape.size());
553 std::transform
554 (
555 shape.cbegin(), shape.cend(),
556 chunk_shape.cbegin(),
557 shape_of_chunks.begin(),
558 [](auto s, auto cs)
559 {
560 std::size_t cn = s / cs;
561 if (s % cs > 0)
562 cn += std::size_t(1); // edge_chunk
563 return cn;
564 }
565 );
566
567 detail::chunk_helper<CS>::resize(m_chunks, shape_of_chunks, chunk_shape, chunk_memory_layout);
568
569 m_shape = xtl::forward_sequence<shape_type, S1>(shape);
570 m_chunk_shape = xtl::forward_sequence<shape_type, S2>(chunk_shape);
571 }

585 std::size_t index_of_chunk = static_cast<size_t>(idx) / m_chunk_shape[dim];

599 template <class CS>
600 template <class T, std::size_t N>
601 inline auto xchunked_array<CS>::unpack(const std::array<T, N> &arr) const -> static_indexes_type<N>
602 {
603 std::array<std::size_t, N> arr0;
604 std::array<std::size_t, N> arr1;
605 for (std::size_t i = 0; i < N; ++i)
606 {
607 arr0[i] = std::get<0>(arr[i]);
608 arr1[i] = std::get<1>(arr[i]);
609 }
610 return std::make_pair(arr0, arr1);
611 }

613 template <class CS>
614 template <class It>
615 inline auto xchunked_array<CS>::get_indexes_dynamic(It first, It last) const -> dynamic_indexes_type
616 {
617 auto size = static_cast<std::size_t>(std::distance(first, last));
618 std::vector<std::size_t> indexes_of_chunk(size);
619 std::vector<std::size_t> indexes_in_chunk(size);
620 for (std::size_t dim = 0; dim < size; ++dim)
621 {
622 auto chunk_index = get_chunk_indexes_in_dimension(dim, *first++);
623 indexes_of_chunk[dim] = chunk_index.first;
624 indexes_in_chunk[dim] = chunk_index.second;
625 }
626 return std::make_pair(indexes_of_chunk, indexes_in_chunk);
627 }
