507 template <class EC, std::size_t N, layout_type L, class Tag>
508 template <class SC>
509 inline xtensor_container<EC, N, L, Tag>& xtensor_container<EC, N, L, Tag>::operator=(xarray_container<EC, L, SC, Tag>&& rhs)
510 {
511 XTENSOR_ASSERT_MSG(N == rhs.dimension(), "Cannot change dimension of xtensor.");
512 std::copy(rhs.shape().begin(), rhs.shape().end(), this->shape_impl().begin());
513 std::copy(rhs.strides().cbegin(), rhs.strides().cend(), this->strides_impl().begin());
514 std::copy(rhs.backstrides().cbegin(), rhs.backstrides().cend(), this->backstrides_impl().begin());
515 this->mutable_layout() = std::move(rhs.layout());
516 m_storage = std::move(std::move(rhs.storage()));
517 return *this;
518 }
