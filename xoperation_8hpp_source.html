32 struct NAME \
33 { \
34 template <class A1> \
35 constexpr auto operator()(const A1& arg) const \
36 { \
37 return OP arg; \
38 } \

40 constexpr auto simd_apply(const B& arg) const \

43 } \

47template <class T1, class T2, XTL_REQUIRES(xtl::negation<std::is_same<T1, T2>>)> \
48constexpr auto operator OP(const std::complex<T1>& arg1, const std::complex<T2>& arg2) \
49{ \
50 using result_type = typename xtl::promote_type_t<std::complex<T1>, std::complex<T2>>; \
51 return (result_type(arg1) OP result_type(arg2)); \
52} \
53 \
54template <class T1, class T2, XTL_REQUIRES(xtl::negation<std::is_same<T1, T2>>)> \
55constexpr auto operator OP(const T1& arg1, const std::complex<T2>& arg2) \
56{ \
57 using result_type = typename xtl::promote_type_t<T1, std::complex<T2>>; \
58 return (result_type(arg1) OP result_type(arg2)); \
59} \
60 \
61template <class T1, class T2, XTL_REQUIRES(xtl::negation<std::is_same<T1, T2>>)> \
62constexpr auto operator OP(const std::complex<T1>& arg1, const T2& arg2) \
63{ \
64 using result_type = typename xtl::promote_type_t<std::complex<T1>, T2>; \
65 return (result_type(arg1) OP result_type(arg2)); \
66}

68#define BINARY_OPERATOR_FUNCTOR(NAME, OP) \
69 struct NAME \
70 { \
71 template <class T1, class T2> \
72 constexpr auto operator()(T1&& arg1, T2&& arg2) const \
73 { \
74 using xt::detail::operator OP; \
75 return (std::forward<T1>(arg1) OP std::forward<T2>(arg2)); \
76 } \
77 template <class B> \
78 constexpr auto simd_apply(const B& arg1, const B& arg2) const \
79 { \
80 return (arg1 OP arg2); \
81 } \

199 {

206 // On MSVC, the second argument of enable_if_t is always evaluated, even if the condition is false.
207 // Wrapping the xfunction type in the xfunction_type metafunction avoids this evaluation when

209 // on vector and arrays iterators.

650 template <class E1, class E2>
651 inline std::enable_if_t<xoptional_comparable<E1, E2>::value, bool>
652 operator==(const xexpression<E1>& e1, const xexpression<E2>& e2)
653 {
654 const E1& de1 = e1.derived_cast();
655 const E2& de2 = e2.derived_cast();
656 bool res = de1.dimension() == de2.dimension() && std::equal(de1.shape().begin(), de1.shape().end(), de2.shape().begin());
657 auto iter1 = de1.begin();
658 auto iter2 = de2.begin();
659 auto iter_end = de1.end();
660 while (res && iter1 != iter_end)
661 {
662 res = (*iter1++ == *iter2++);
663 }
664 return res;
665 }

814 template <>
815 struct next_idx_impl<layout_type::row_major>
816 {
817 template <class S, class I>
818 inline auto operator()(const S& shape, I& idx)
819 {
820 for (std::size_t j = shape.size(); j > 0; --j)
821 {
822 std::size_t i = j - 1;
823 if (idx[i] >= shape[i] - 1)
824 {
825 idx[i] = 0;
826 }
827 else
828 {
829 idx[i]++;
830 return idx;
831 }
832 }
833 // return empty index, happens at last iteration step, but remains unused
834 return I();
835 }
836 };

877 template <class T>
878 inline auto nonzero(const T& arr)
879 {
880 auto shape = arr.shape();
881 using index_type = xindex_type_t<typename T::shape_type>;
882 using size_type = typename T::size_type;
883
884 auto idx = xtl::make_sequence<index_type>(arr.dimension(), 0);
885 std::vector<std::vector<size_type>> indices(arr.dimension());
886
887 size_type total_size = compute_size(shape);
888 for (size_type i = 0; i < total_size; i++, detail::next_idx(shape, idx))
889 {
890 if (arr.element(std::begin(idx), std::end(idx)))
891 {
892 for (std::size_t n = 0; n < indices.size(); ++n)
893 {
894 indices.at(n).push_back(idx[n]);
895 }
896 }
897 }
898
899 return indices;
900 }
