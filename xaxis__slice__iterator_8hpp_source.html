143 template <class CT>
144 template <class CTA>
145 inline xaxis_slice_iterator<CT>::xaxis_slice_iterator(CTA&& e, size_type axis, size_type index, size_type offset) :
146 p_expression(get_storage_init<storing_type>(std::forward<CTA>(e))), m_index(index),
147 m_offset(offset), m_axis_stride(static_cast<size_type>(e.strides()[axis]) * (e.shape()[axis] - 1u)),
148 m_lower_shape(0), m_upper_shape(0), m_iter_size(0), m_is_target_axis(false),
149 m_sv(strided_view(std::forward<CT>(e), std::forward<shape_type>({ e.shape()[axis] }),
150 std::forward<strides_type>({ e.strides()[axis] }), offset, e.layout()))
151 {
152 if (e.layout() == layout_type::row_major)
153 {
154 m_is_target_axis = axis == e.dimension() - 1;
155 m_lower_shape = std::accumulate(e.shape().begin() + axis + 1, e.shape().end(), size_t(1), std::multiplies<>());
156 m_iter_size = std::accumulate(e.shape().begin() + 1, e.shape().end(), size_t(1), std::multiplies<>());
157
158 }
159 else
160 {
161 m_is_target_axis = axis == 0;
162 m_lower_shape = std::accumulate(e.shape().begin(), e.shape().begin() + axis, size_t(1), std::multiplies<>());
163 m_iter_size = std::accumulate(e.shape().begin(), e.shape().end() - 1, size_t(1), std::multiplies<>());
164 }
165 m_upper_shape = m_lower_shape + m_axis_stride;
166 }

176 template <class CT>
177 inline auto xaxis_slice_iterator<CT>::operator++() -> self_type&
178 {
179 ++m_index; ++m_offset;
180 auto index_compare = (m_offset % m_iter_size);
181 if (m_is_target_axis || (m_upper_shape >= index_compare && index_compare >= m_lower_shape))
182 {
183 m_offset += m_axis_stride;
184 }
185 m_sv.set_offset(m_offset);
186 return *this;
187 }
